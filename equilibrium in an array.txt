#include <stdio.h>

int main() {
    int n,a[100],l=0,r=0,i,j,k;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=1;i<n;i++){
        l=0;
        r=0;
        for(j=0;j<i;j++){
            l=l+a[j];
            
        }
        for(k=i+1;k<n;k++){
            r=r+a[k];
        }
        if(l==r){
            
            printf("Equilibrium element:%d\n",a[i]);
            printf("Equilibrium index:%d",i+1);
            break;
        }
        else{
            continue;
        }
    }
    return 0;
}
