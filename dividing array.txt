#include<stdio.h>
int main(){
    int a[100],n,b,i;
    scanf("%d",&n);
    
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    if(n%2==0){
        b=n/2;
        printf("Sub array 1:\n");
        for(i=0;i<b;i++){
            printf("%d\n",a[i]);
        }
        printf("Sub array 2:\n");
        for(i=b;i<n;i++){
            printf("%d\n",a[i]);
        }
    }
    else{
        printf("The given array cannot be divided");
    }
    return 0;
}
