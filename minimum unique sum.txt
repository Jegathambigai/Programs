
#include <stdio.h>

int main() {
    int a[100],n,i,j,t,b=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        for(j=i+1;j<n;j++){
            if(a[i]>a[j]){
              t=a[i];
              a[i]=a[j];
              a[j]=t;
            }
        }
    }
    for(i=0;i<n;i++){
        if(a[i]!=a[i+1]){
            b=b+a[i];
            printf("%d ",a[i]);
        }
    }
    printf("Minimum unique sum:%d",b);
    return 0;
}
