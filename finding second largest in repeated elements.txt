#include<stdio.h>
int main(){
    int n,a[100],i,t,b[100],c=0,d=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    
    for(i=0;i<n;i++){
        for(int j=i+1;j<n;j++){
        if(a[i]>a[j]){
            t=a[i];
            a[i]=a[j];
            a[j]=t;
        }
        else{
            continue;
        }
        } 
        
    }
   
    for(i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(a[i]==a[j]){
                if(i==j){
                    b[c++]=a[i];
                    d=d+1;
                }
                else if(i>j){
                    break;
                }
                else if(i<j){
                    break;
                }
            }
        }
    }
    
    printf("%d",a[d-2]);
    return 0;
}
