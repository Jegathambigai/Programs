#include<stdio.h>
int main(){
int a[100],n,b[2],t,i,j,a=0;
printf("Enter the array size:");
scanf("%d",&n);
printf("Enter the array elements:");
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
}
printf("Enter the target element:");
scanf("%d",t);
for(i=0;i<n;i++){
for(j=i+1;j<n;j++){
if((a[i]+a[j])==t){
b[a++]=i;
b[a++]=j;
break;
}
}
if(a==2){
break;
}
printf("%d,%d",b[0],b[1]);
return 0;
}
