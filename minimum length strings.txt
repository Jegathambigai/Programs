#include<stdio.h>
int main(){
    int i,j,n,c=0,b[100],d=0;
    char a[100][100];
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%s",&a[i]);
    }
    for(i=0;i<n;i++){
        for(j=0;a[i][j]!='\0';j++){
            c=c+1;
        }b[i]=c;
        c=0;
    }
    for(i=0;i<n;i++){
        if(b[i]<b[i+1]){
            d=b[i];
        }
    }
    printf("Minimum length strings:\n");
    for(i=0;i<n;i++){
        if(b[i]==d){
            printf("%s\n",a[i]);
        }
    }
}
