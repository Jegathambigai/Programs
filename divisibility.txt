#include<stdio.h>
int main(){
    int n,a[100],b,i,c=1,j;
    long int e=1,d=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
    }
    for(i=0;i<n;i++){
        
        c=a[i]%10;
		for(j=1;j<=(n-i-1);j++){
			e=e*10;
		}
		
        d=d+(c*e);
        
		c=1;
		e=1;
        }
       
        
        
    if(d%10==0){
        printf("Yes");
    }
    else{
        printf("No");
    }
    return 0;
}
